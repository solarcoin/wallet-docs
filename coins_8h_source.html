168  template<typename Stream>
169  void Serialize(Stream &s) const {
170  unsigned int nMaskSize = 0, nMaskCode = 0;
171  CalcMaskSize(nMaskSize, nMaskCode);
172  bool fFirst = vout.size() > 0 && !vout[0].IsNull();
173  bool fSecond = vout.size() > 1 && !vout[1].IsNull();
174  assert(fFirst || fSecond || nMaskCode);
175  unsigned int nCode = 8*(nMaskCode - (fFirst || fSecond ? 0 : 1)) + (fCoinBase ? 1 : 0) + (fFirst ? 2 : 0) + (fSecond ? 4 : 0);
176  // version
177  ::Serialize(s, VARINT(this->nVersion));
178  // header code
179  ::Serialize(s, VARINT(nCode));
180  // spentness bitmask
181  for (unsigned int b = 0; b<nMaskSize; b++) {
182  unsigned char chAvail = 0;
183  for (unsigned int i = 0; i < 8 && 2+b*8+i < vout.size(); i++)
184  if (!vout[2+b*8+i].IsNull())
185  chAvail |= (1 << i);
186  ::Serialize(s, chAvail);
187  }
188  // txouts themself
189  for (unsigned int i = 0; i < vout.size(); i++) {
190  if (!vout[i].IsNull())
191  ::Serialize(s, CTxOutCompressor(REF(vout[i])));
192  }
193  // coinbase height
194  ::Serialize(s, VARINT(nHeight));
195  }

197  template<typename Stream>
198  void Unserialize(Stream &s) {
199  unsigned int nCode = 0;
200  // version
201  ::Unserialize(s, VARINT(this->nVersion));
202  // header code
203  ::Unserialize(s, VARINT(nCode));
204  fCoinBase = nCode & 1;
205  std::vector<bool> vAvail(2, false);
206  vAvail[0] = (nCode & 2) != 0;
207  vAvail[1] = (nCode & 4) != 0;
208  unsigned int nMaskCode = (nCode / 8) + ((nCode & 6) != 0 ? 0 : 1);
209  // spentness bitmask
210  while (nMaskCode > 0) {
211  unsigned char chAvail = 0;
212  ::Unserialize(s, chAvail);
213  for (unsigned int p = 0; p < 8; p++) {
214  bool f = (chAvail & (1 << p)) != 0;
215  vAvail.push_back(f);
216  }
217  if (chAvail != 0)
218  nMaskCode--;
219  }
220  // txouts themself
221  vout.assign(vAvail.size(), CTxOut());
222  for (unsigned int i = 0; i < vAvail.size(); i++) {
223  if (vAvail[i])
224  ::Unserialize(s, REF(CTxOutCompressor(vout[i])));
225  }
226  // coinbase height
227  ::Unserialize(s, VARINT(nHeight));
228  Cleanup();
229  }
