477  inline void SerializationOp(Stream& s, Operation ser_action) {
478  int nVersion = s.GetVersion();
479  if (!(s.GetType() & SER_GETHASH))
480  READWRITE(VARINT(nVersion));

486  READWRITE(VARINT(nFile));

494  READWRITE(VARINT(nMoneySupply));
495  READWRITE(VARINT(nFlags));
496  READWRITE(VARINT(nStakeModifier));
497  if (IsProofOfStake()) {
498  READWRITE(prevoutStake);
499  READWRITE(nStakeTime);

501  } else {

505  const_cast<CDiskBlockIndex*>(this)->hashProofOfStake = uint256();
506  }
507  }

510  READWRITE(this->nVersion);
511  READWRITE(hashPrev);
512  READWRITE(hashMerkleRoot);
513  READWRITE(nTime);
514  READWRITE(nBits);
